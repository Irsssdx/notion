2. Fetch Api Method using fetch().then().then().catch()

# GET METHOD
fetch(URL)
.then((res) => res.json())
.then((json_data) => {
    console.log("JSON_DATA = ", json_data);
    setData(json_data)
})
.catch((err) => {
    console.log("Error came = ", err);
})
# PUT & POST METHOD
const data = {
    id: "def234",
    name: "alan", 
    age: 21,
}

const URL = data.id ? "https://jsonplaceholder.typicode.com/comments/" + data.id : "https://jsonplaceholder.typicode.com/comments";

fetch(URL, {
    method: data.id ? "PUT" : "POST",
    headers: {
        "Content-Type": "application/json",
        "Access-Control-Allow-Origin": "*"
    },
    body: JSON.stringify(data)
})
.then((res) => res.json())
.then((json_data) => {
    console.log("JSON_DATA = ", json_data);
    setData(json_data)
})
.catch((err) => {
    console.log("Error came = ", err);
})
